function cache(func) {
  const cache = {}
  return function(...args) {
    const arg = JSON.stringify(args)
    if(!(arg in cache)) {
      const result = func(...args)
      cache[arg] = result
    }
    return cache[arg]
  }
}
                
function getCount(str) {
  let vowelsCount = 0;
  const vowStr = 'aeiou'
  str.split('').forEach(vow => {
    if(vowStr.includes(vow)) vowelsCount++
  })
  return vowelsCount;
}
